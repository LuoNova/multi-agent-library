# 1. 预约书籍接口重置数据库sql

## -- ------阶段1，测试本馆没有书，需要调配------ --

-- 1. 重置图书副本表（理科馆0本AVAILABLE，文科馆2本AVAILABLE）
UPDATE tb_book_copy SET
    library_id = 2,
    status = 'AVAILABLE',
    local_borrow_count = 0,
    last_borrow_time = NULL,
    update_time = NOW()
WHERE id IN (1001, 1002);

-- 2. 重置馆藏书目统计表
UPDATE tb_library_biblio_stats SET
    stock_count = 0,
    borrow_count_30d = 0,
    reservation_pending_count = 0
WHERE library_id = 1 AND biblio_id = 1;

UPDATE tb_library_biblio_stats SET
    stock_count = 2,
    borrow_count_30d = 5,
    reservation_pending_count = 0
WHERE library_id = 2 AND biblio_id = 1;

-- 3. 清空调拨记录
DELETE FROM tb_book_transfer WHERE copy_id IN (1001, 1002);

-- 4. 清空借阅记录（避免"已借阅该书"的限制）
DELETE FROM tb_book_borrow;

-- 5. 重置用户借阅计数
UPDATE tb_user SET current_borrow_count = 0 WHERE id = 1;

-- 验证初始状态
SELECT '副本状态' as check_item;
SELECT id, library_id, status FROM tb_book_copy WHERE biblio_id = 1;
SELECT '统计表' as check_item;
SELECT library_id, stock_count FROM tb_library_biblio_stats WHERE biblio_id = 1;

## -- ------阶段2，测试本地有书，直接借书------ --

-- 手动将第二本书调到理科馆（模拟之前已完成调拨）
UPDATE tb_book_copy SET library_id = 1, status = 'AVAILABLE' WHERE id = 1002;

-- 更新统计表（理科馆2本，文科馆0本）
UPDATE tb_library_biblio_stats SET stock_count = 2 WHERE library_id = 1 AND biblio_id = 1;
UPDATE tb_library_biblio_stats SET stock_count = 0 WHERE library_id = 2 AND biblio_id = 1;

-- 先清空之前的借阅记录（测试用）
DELETE FROM tb_book_borrow WHERE user_id = 1;
UPDATE tb_user SET current_borrow_count = 0 WHERE id = 1;

## -- ------阶段3，测试全图书馆都没有书，需要进行预约------ --

-- 将两本书都设为已借出（模拟全馆无库存）
UPDATE tb_book_copy SET status = 'BORROWED' WHERE id IN (1001, 1002);

-- 或者设为其他状态，只要不是 AVAILABLE
-- 确保统计表也同步（可选）
UPDATE tb_library_biblio_stats SET stock_count = 0 WHERE biblio_id = 1;

-- 准备测试数据（还书触发预约场景）
-- 假设：用户1（id=1）已预约《Java编程》，用户2（id=2）借了这本书并来还书

-- 1. 确保用户2存在（还书人）
INSERT IGNORE INTO tb_user (id, student_no, name, max_borrow_count, current_borrow_count, status) 
VALUES (2, '2024002', '还书测试用户', 10, 1, 'ACTIVE');

-- 2. 重置副本1001状态：模拟被用户2借走，在理科馆（id=1）
UPDATE tb_book_copy SET 
    library_id = 1,           -- 当前在理科馆
    status = 'BORROWED',       -- 已借出状态
    update_time = NOW()
WHERE id = 1001;

-- 3. 创建用户2的借阅记录（未归还）
DELETE FROM tb_book_borrow WHERE copy_id = 1001; -- 清理旧数据
INSERT INTO tb_book_borrow (copy_id, user_id, borrow_time, due_time, status, create_time) 
VALUES (1001, 2, NOW() - INTERVAL 5 DAY, NOW() + INTERVAL 25 DAY, 'BORROWING', NOW());

-- 4. 创建用户1的预约记录（PENDING状态，希望在理科馆取书）
DELETE FROM tb_book_reservation WHERE user_id = 1 AND biblio_id = 1; -- 清理旧数据
INSERT INTO tb_book_reservation (biblio_id, user_id, pickup_library_id, copy_id, reserve_time, expire_time, status, notification_sent, create_time)
VALUES (1, 1, 1, NULL, NOW(), NOW() + INTERVAL 3 DAY, 'PENDING', 0, NOW());

-- 5. 同步统计表（理科馆这本被借走了，可用库存应为0，但还书后会恢复）
UPDATE tb_library_biblio_stats SET stock_count = 0 WHERE library_id = 1 AND biblio_id = 1;

-- 6. 确认用户1当前借阅计数（确保其能再借书）
UPDATE tb_user SET current_borrow_count = 0 WHERE id = 1;

-- 验证数据
SELECT '副本状态' as check_item, id, library_id, status, biblio_id FROM tb_book_copy WHERE id = 1001;
SELECT '借阅记录' as check_item, user_id, status FROM tb_book_borrow WHERE copy_id = 1001;
SELECT '预约记录' as check_item, user_id, status, pickup_library_id FROM tb_book_reservation WHERE biblio_id = 1;
