-- 重置图书副本表（关键：理科馆0本，文科馆2本，状态重置为可借）
UPDATE tb_book_copy SET
    library_id = 2,           -- 回到文科馆
    status = 'AVAILABLE',      -- 重置为可借状态
    update_time = NOW()
WHERE biblio_id = 1;          -- 针对《Java编程》这本书

-- 如果之前测试产生了新的调拨记录，可能需要指定具体的copy_id重置：
UPDATE tb_book_copy SET library_id = 2, status = 'AVAILABLE' WHERE id IN (1001, 1002);

-- 重置馆藏书目统计表（关键：同步库存数量）
UPDATE tb_library_biblio_stats SET
    stock_count = 0,          -- 理科馆清0
    borrow_count_30d = 0,
    reservation_pending_count = 0
WHERE library_id = 1 AND biblio_id = 1;

UPDATE tb_library_biblio_stats SET
    stock_count = 2,          -- 文科馆恢复2本
    borrow_count_30d = 5,
    reservation_pending_count = 0
WHERE library_id = 2 AND biblio_id = 1;
